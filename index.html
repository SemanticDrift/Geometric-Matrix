<script>

    const fieldIds = ['input_a', 'input_b', 'input_c', 'input_d'];


    fieldIds.forEach(id => {

        document.getElementById(id).addEventListener('input', solveAndDraw);

    });


    function fmt(n) {

        if (n === null || n === undefined || isNaN(n)) return '';

        if (Number.isInteger(n)) return String(n);

        const r = parseFloat(n.toFixed(4));

        return Number.isInteger(r) ? String(r) : r.toString();

    }


    function setDerived(id, val) {

        document.getElementById(id).value = fmt(val);

    }


    function getRaw(id) {

        const v = parseFloat(document.getElementById(id).value);

        return isNaN(v) ? null : v;

    }


    function showError() {

        document.getElementById('deltaVal').innerHTML =

            '<span class="red">ERROR</span>';

        document.getElementById('typeName').innerText = '';

        clearCanvas();

    }


    function clearCanvas() {

        const ctx = document.getElementById('sspCanvas').getContext('2d');

        ctx.clearRect(0, 0, 520, 340);

    }


    function clearDisplay() {

        document.getElementById('deltaVal').innerText = '';

        document.getElementById('typeName').innerText = '';

        clearCanvas();

    }


    function isValid(a, b, c, d) {

        if (a <= 0 || b <= 0 || c <= 0 || d <= 0) return false;

        if (a >= d) return false;

        return true;

    }


    function solveAndDraw() {

        let a = getRaw('input_a');

        let b = getRaw('input_b');

        let c = getRaw('input_c');

        let d = getRaw('input_d');


        const filled = [a,b,c,d].filter(v => v !== null).length;

        if (filled < 2) { clearDisplay(); return; }


        // Derive from any valid pair — a,d is primary

        if (a !== null && d !== null) {

            b = (a + d) / 2;

            c = d - a;

            setDerived('input_b', b);

            setDerived('input_c', c);

        } else if (b !== null && c !== null) {

            a = b - c / 2;

            d = b + c / 2;

            setDerived('input_a', a);

            setDerived('input_d', d);

        } else if (a !== null && b !== null) {

            d = 2 * b - a;

            c = d - a;

            setDerived('input_d', d);

            setDerived('input_c', c);

        } else if (a !== null && c !== null) {

            d = a + c;

            b = (a + d) / 2;

            setDerived('input_d', d);

            setDerived('input_b', b);

        } else if (c !== null && d !== null) {

            a = d - c;

            b = (a + d) / 2;

            setDerived('input_a', a);

            setDerived('input_b', b);

        } else if (b !== null && d !== null) {

            a = 2 * b - d;

            c = d - a;

            setDerived('input_a', a);

            setDerived('input_c', c);

        } else {

            clearDisplay();

            return;

        }


        if (!isValid(a, b, c, d)) { showError(); return; }


        const delta = b*b + c*c - d*d;

        const deltaEl = document.getElementById('deltaVal');

        const typeEl = document.getElementById('typeName');


        deltaEl.innerText = fmt(delta);

        typeEl.className = 'triangle-name';


        const isLocked = Math.abs(delta) < 0.001;

        const isObtuse = delta < -0.001;


        if (isLocked) {

            typeEl.innerText = 'RECTANGULO';

            typeEl.classList.add('green');

        } else if (isObtuse) {

            typeEl.innerText = 'OBTUSANGULO';

            typeEl.classList.add('orange');

        } else {

            typeEl.innerText = 'ACUTANGULO';

            typeEl.classList.add('blue');

        }


        drawTriangle(b, c, d, delta, isLocked, isObtuse);

    }


    function drawTriangle(b, c, d, delta, isLocked, isObtuse) {

        const canvas = document.getElementById('sspCanvas');

        const ctx = canvas.getContext('2d');

        const W = canvas.width;

        const H = canvas.height;


        ctx.clearRect(0, 0, W, H);


        // Angle at vertex A (left base) between sides b and c, opposite d

        const cosA = (b*b + c*c - d*d) / (2*b*c);

        const sinA = Math.sqrt(Math.max(0, 1 - cosA*cosA));

        if (sinA < 0.0001) return;


        const margin = 60;

        const maxW = W - margin * 2;

        const maxH = H - margin - 40;

        const triH = b * sinA;

        const scale = Math.min(maxW / c, maxH / triH, 45);


        const baseLen = c * scale;

        const xA = (W - baseLen) / 2;

        const xD = xA + baseLen;

        const baseY = H - margin;


        // Apex using angle at A

        const apexX = xA + b * cosA * scale;

        const apexY = baseY - b * sinA * scale;


        const color = isLocked ? '#1a7a1a' : isObtuse ? '#bb5500' : '#1144bb';


        // Fill

        ctx.beginPath();

        ctx.moveTo(xA, baseY);

        ctx.lineTo(xD, baseY);

        ctx.lineTo(apexX, apexY);

        ctx.closePath();

        ctx.fillStyle = color + '25';

        ctx.fill();


        // Stroke

        ctx.beginPath();

        ctx.moveTo(xA, baseY);

        ctx.lineTo(xD, baseY);

        ctx.lineTo(apexX, apexY);

        ctx.closePath();

        ctx.strokeStyle = color;

        ctx.lineWidth = 2.5;

        ctx.stroke();


        // Right angle box at the base left corner (xA, baseY)

        // This is where the 90 degree angle is for the 3-4-5 triangle

        if (isLocked) {

            const sz = 13;

            ctx.beginPath();

            ctx.moveTo(xA + sz, baseY);

            ctx.lineTo(xA + sz, baseY - sz);

            ctx.lineTo(xA, baseY - sz);

            ctx.strokeStyle = '#1a7a1a';

            ctx.lineWidth = 1.5;

            ctx.stroke();

        }


        // Vertex labels

        ctx.fillStyle = '#888';

        ctx.font = '12px sans-serif';

        ctx.fillText('a', xA - 14, baseY + 5);

        ctx.fillText('d', xD + 5, baseY + 5);


        // Side labels

        ctx.fillStyle = color;

        ctx.font = 'bold 11px sans-serif';

        ctx.fillText('c = ' + fmt(c), (xA+xD)/2 - 18, baseY + 18);

        ctx.fillText('b = ' + fmt(b), (xA+apexX)/2 - 32, (baseY+apexY)/2 - 2);

        ctx.fillText('d = ' + fmt(d), (xD+apexX)/2 + 4, (baseY+apexY)/2 - 2);

    }
